################################################################################
# Compile-time probes
# Each .cpp forces static constexpr filter construction; compile time IS the
# benchmark. OBJECT lib = compile only, no link step.
################################################################################
file(GLOB CT_SOURCES "compile_time/profile_*.cpp")
foreach(src ${CT_SOURCES})
    get_filename_component(tgt ${src} NAME_WE)
    add_library(${tgt} OBJECT ${src})
    target_link_libraries(${tgt} PRIVATE constfilt constfilt_warnings)
endforeach()

################################################################################
# Runtime benchmark executable
################################################################################
# bench_accuracy: constfilt vs Octave reference (C++17, no external deps)
# accuracy_reference.hpp is generated by:
#   octave --no-gui profiling/octave/generate_accuracy_reference.m
if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/accuracy_reference.hpp")
    add_executable(bench_accuracy bench_accuracy.cpp)
    target_include_directories(bench_accuracy PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR})
    target_link_libraries(bench_accuracy PRIVATE constfilt)
    set_target_properties(bench_accuracy PROPERTIES
        CXX_STANDARD 17 CXX_STANDARD_REQUIRED ON)
    if(TARGET iir)
        target_link_libraries(bench_accuracy PRIVATE iir)
        target_compile_definitions(bench_accuracy PRIVATE CONSTFILT_BENCH_IIR1)
    endif()

    # bench_accuracy_kfr: KFR-only accuracy (C++20, same split as bench_kfr)
    if(TARGET kfr_dsp)
        add_executable(bench_accuracy_kfr bench_accuracy_kfr.cpp)
        target_include_directories(bench_accuracy_kfr PRIVATE
            ${CMAKE_CURRENT_SOURCE_DIR})
        target_link_libraries(bench_accuracy_kfr PRIVATE kfr_dsp)
        set_target_properties(bench_accuracy_kfr PROPERTIES
            CXX_STANDARD 20 CXX_STANDARD_REQUIRED ON)
    elseif(TARGET kfr)
        add_executable(bench_accuracy_kfr bench_accuracy_kfr.cpp)
        target_include_directories(bench_accuracy_kfr PRIVATE
            ${CMAKE_CURRENT_SOURCE_DIR})
        target_link_libraries(bench_accuracy_kfr PRIVATE kfr)
        set_target_properties(bench_accuracy_kfr PROPERTIES
            CXX_STANDARD 20 CXX_STANDARD_REQUIRED ON)
    endif()
else()
    message(STATUS
        "profiling/accuracy_reference.hpp not found — skipping bench_accuracy. "
        "Run: octave --no-gui profiling/octave/generate_accuracy_reference.m")
endif()

# bench: constfilt + iir1 (C++17)
# KFR headers require C++20 but consteig vendor headers use C++17 aggregate
# initialisation that breaks under GCC 15 in C++20 mode. Keep them in separate
# compilation units.
add_executable(bench bench.cpp)
target_link_libraries(bench PRIVATE constfilt)
set_target_properties(bench PROPERTIES CXX_STANDARD 17 CXX_STANDARD_REQUIRED ON)

if(TARGET iir)
    target_link_libraries(bench PRIVATE iir)
    target_compile_definitions(bench PRIVATE CONSTFILT_BENCH_IIR1)
endif()

# bench_kfr: KFR only (C++20); appends rows to the same runtime CSV

    target_link_libraries(bench_kfr PRIVATE kfr_dsp)
    set_target_properties(bench_kfr PROPERTIES
        CXX_STANDARD 20 CXX_STANDARD_REQUIRED ON)
elseif(TARGET kfr)
    add_executable(bench_kfr bench_kfr.cpp)
    target_link_libraries(bench_kfr PRIVATE kfr)
    set_target_properties(bench_kfr PROPERTIES
        CXX_STANDARD 20 CXX_STANDARD_REQUIRED ON)
endif()
